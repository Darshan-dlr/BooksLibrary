def test_add_book(client):
    """Test adding a new book."""
    new_book = {
        "title": "New Book",
        "author": "Author Name",
        "genre": ["Fiction"],
        "year_published": 2022,
        "summary": "A new fictional book.",
    }
    response = client.post("/books", json=new_book)
    assert response.status_code == 201
    assert "id" in response.json()
